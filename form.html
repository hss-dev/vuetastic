The Mighty Vue

Tag prop: "{{value}}"

Send Event back to angular on change when field more than 3 characters... 
const VComponent = Vue.component('hello-component', {
props: {
    start: String
},
computed: {
    value: function() {
        return this.start;
    }
},
methods: {
    bang: function() {
        console.log("Vue's been clicked");
    },
    updateValue: function(value) {
        console.log(value);
        if (value.length > 3) {
            console.log("SENDING event");
            const args = {
                detail: {
                    hello: value
                }
            };
            window.dispatchEvent(
                new CustomEvent('valid', args));
        }
    }
},
template
});
